% ================= 2. Snake 方程（Yanking） =================
\begin{scope}[shift={(0,-6)}]
  \node[label, anchor=south] at (0,3.2) {左 Snake：$(\mathrm{id}_A \otimes \epsilon_A) \circ (\eta_A \otimes \mathrm{id}_A) \cong \mathrm{id}_A$};

  % 弯曲版本
  \draw[wire] (0,2.8) -- (0,1.6);
  \draw[wire] (0,1.6) .. controls (0,0.8) and (1.2,0.8) .. (1.2,0)
    node[right, pos=0.5] {$A^*$};
  \draw[wire] (1.2,0) .. controls (1.2,-0.8) and (0,-0.8) .. (0,-1.6);
  \draw[wire] (0,-1.6) -- (0,-2.8);
  \node[label] at (0,3.0) {$A$};
  \node[label] at (0,-3.0) {$A$};

  \node[label, font=\Large] at (2.8,0) {$\cong$};

  % 拉直版本
  \draw[wire] (6,2.8) -- (6,-2.8);
  \node[label] at (6,3.0) {$A$};
  \node[label] at (6,-3.0) {$A$};
  \node[label, anchor=south] at (6,3.2) {恒等 $\mathrm{id}_A$};
\end{scope}

% ================= 3. 右 Snake 方程 =================
\begin{scope}[shift={(0,-14)}]
  \node[label, anchor=south] at (0,3.2) {右 Snake：$(\epsilon_A \otimes \mathrm{id}_{A^*}) \circ (\mathrm{id}_{A^*} \otimes \eta_A) \cong \mathrm{id}_{A^*}$};

  \node[label] at (0,3.0) {$A^*$};
  \node[label] at (0,-3.0) {$A^*$};
